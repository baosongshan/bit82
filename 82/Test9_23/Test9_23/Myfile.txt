#include<stdio.h>

int main()
{
	FILE *fpIn, *fpOut;
	fpIn = fopen("Test.c", "r");
	if (NULL == fpIn)
	{
		printf("Open Test.c Error.\n");
		return -1;
	}
	fpOut = fopen("Myfile.txt", "w");
	if (NULL == fpOut)
	{
		fclose(fpIn);
		printf("Open Myfile.txt Error.\n");
		return -1;
	}

	char ch;
	while (!feof(fpIn))
	{
		ch = fgetc(fpIn);
		fputc(ch, fpOut);
	}

	fclose(fpIn);
	fclose(fpOut);
	return 0;
}
